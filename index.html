R Script The demonstrated color analysis of two before and after screenshots library(png) library(grid) library(gridExtra) library(treemap) i <- readPNG("stage2Before.png") dfi = data.frame( red = matrix(i[,,1], ncol=1), green = matrix(i[,,2], ncol=1), blue = matrix(i[,,3], ncol=1) ) j <- readPNG("stage2After.png") dfj = data.frame( red = matrix(j[,,1], ncol=1), green = matrix(j[,,2], ncol=1), blue = matrix(j[,,3], ncol=1) ) ti <- as.data.frame(table(rgb(dfi$red, dfi$green, dfi$blue))) tj <- as.data.frame(table(rgb(dfj$red, dfj$green, dfj$blue))) k <- merge(ti, tj, by="Var1") k$dif <- k$Freq.y - k$Freq.x l <- k[k$dif > 0,] png("col2P.png") treemap( l, index=c("Var1", "dif"), vSize="dif", vColor="Var1", type="color", fontsize.labels=(0), border.lwds=(0) ) dev.off() m <- k[k$dif < 0,] m$dif <- m$dif * -1 png("col2M.png") treemap( m, index=c("Var1", "dif"), vSize="dif", vColor="Var1", type="color", fontsize.labels=(0), border.lwds=(0) ) dev.off() c <- table(rgb(dfi$red,dfi$green,dfi$blue)) c <- as.data.frame(c) c[,2] <- as.numeric(as.character(c[,2])) png("col2B.png") treemap( c, index=c("Var1"), vSize="Freq", vColor="Var1", type="color", fontsize.labels=(0), border.lwds=(0) ) dev.off() c <- table(rgb(dfj$red,dfj$green,dfj$blue)) c <- as.data.frame(c) c[,2] <- as.numeric(as.character(c[,2])) png("col2A.png") treemap( c, index=c("Var1"), vSize="Freq", vColor="Var1", type="color", fontsize.labels=(0), border.lwds=(0) ) dev.off()
